\section{Algorithm Pseudocode}
\label{app:algorithms}

This appendix provides pseudocode for the core algorithms of the \Apeiron{} Framework, focusing on the multi-axial atomicity test and the falsification experiment.

% Define the \Assert command for use in algorithmicx
\algnewcommand{\Assert}[1]{\State \textbf{assert} #1}

\begin{algorithm}[H]
\caption{Multi-Axial Atomicity Test}
\label{alg:atomicity}
\begin{algorithmic}[1]
\Function{IsAtomic}{$o$, $\meta$}
    \State $\text{scores} \gets \text{empty dictionary}$
    \For{$f \in \{\text{bool}, \text{meas}, \text{cat}, \text{info}\}$}
        \State $\Delta_f \gets \text{GetDecompositionOperator}(f)$
        \If{$\Delta_f.\text{can\_decompose}(o)$}
            \State $\text{parts} \gets \Delta_f.\text{decompose}(o)$
            \If{$\text{parts} = \{o\}$}
                \State $\text{scores}[f] \gets 1.0$
            \Else
                \State $\text{scores}[f] \gets 0.0$
            \EndIf
        \Else
            \State $\text{scores}[f] \gets 1.0$ \Comment{not applicable $\Rightarrow$ atomic}
        \EndIf
    \EndFor
    \State \Return $\sum_{f} \meta.\text{weights}[f] \cdot \text{scores}[f]$
\EndFunction
\end{algorithmic}
\end{algorithm}

Algorithm~\ref{alg:atomicity} computes the combined atomicity score for a given observable $o$ using the decomposition operators registered in the framework. The \texttt{MetaSpecification} $\meta$ provides the weights for each framework. The algorithm returns a score in $[0,1]$, with values near $1$ indicating high confidence that $o$ is irreducible.

\begin{algorithm}[H]
\caption{Falsification Experiment for Layer~1}
\label{alg:falsification}
\begin{algorithmic}[1]
\Procedure{RunFalsificationExperiment}{}
    \State $\text{obs\_one} \gets \text{UltimateObservable}(1, \text{DISCRETE})$
    \State $\text{bool\_heuristic} \gets \text{boolean\_atomicity}(\text{obs\_one})$
    \State $\text{bool\_formal} \gets \text{decomposition\_boolean\_atomicity}(\text{obs\_one})$
    \State $\text{meas\_formal} \gets \text{decomposition\_measure\_atomicity}(\text{obs\_one})$
    \Assert{$\text{bool\_heuristic} = 1.0$}
    \Assert{$\text{bool\_formal} = 1.0$}
    \Assert{$\text{meas\_formal} \ge 0.99$}
    \Statex
    \State $\text{obs\_info} \gets \text{UltimateObservable}(\text{``1''} \times 200, \text{RELATIONAL})$
    \State $\text{info\_score} \gets \text{info\_atomicity}(\text{obs\_info})$
    \Assert{$\text{info\_score} \ge 0.99$}
    \Statex
    \State \Return $\text{true}$
\EndProcedure
\end{algorithmic}
\end{algorithm}

Algorithm~\ref{alg:falsification} implements the falsification experiment described in Section~\ref{sec:validation}. The procedure tests the multi-axial convergence hypothesis by verifying that the integer $1$ is recognized as atomic by the Boolean and measure-theoretic frameworks, and that a long repetitive string is recognized as information-theoretically atomic. All assertions must pass for the experiment to confirm the hypothesis.

\begin{algorithm}[H]
\caption{Relational Influence Computation (DensityField)}
\label{alg:density}
\begin{algorithmic}[1]
\Function{ComputeInfluence}{$\text{target}$, $\text{field}$}
    \State $\text{influence} \gets \text{empty dictionary}$
    \For{$\text{other} \in \text{field} \setminus \{\text{target}\}$}
        \State $\text{sim} \gets \text{CosineSimilarity}(\text{target}.\text{embedding}, \text{other}.\text{embedding})$
        \State $\text{dist} \gets 1 - \text{sim}$
        \State $\text{res\_infl} \gets 0$
        \For{$\text{layer} \in \text{target}.\text{resonance\_map} \cap \text{other}.\text{resonance\_map}$}
            \State $w_t \gets \text{target}.\text{resonance\_map}[\text{layer}].\text{weight}$
            \State $w_o \gets \text{other}.\text{resonance\_map}[\text{layer}].\text{weight}$
            \State $\text{res\_infl} \gets \text{res\_infl} + \text{decay} \cdot w_t \cdot w_o / (1 + \text{dist})$
        \EndFor
        \State $\text{emb\_infl} \gets \text{decay} \cdot \max(0, \text{sim})$
        \State $\text{total} \gets \alpha \cdot \text{res\_infl} + (1-\alpha) \cdot \text{emb\_infl}$
        \State $\text{influence}[\text{other}.\text{perspective}] \gets \text{influence}[\text{other}.\text{perspective}] + \text{total}$
    \EndFor
    \State \Return $\text{influence}$
\EndFunction
\end{algorithmic}
\end{algorithm}

Algorithm~\ref{alg:density} sketches the two-channel influence computation in the \texttt{DensityField} class of Layer~2. The function computes the cumulative influence of all other observables on a target observable, aggregating by observer perspective. The resonance channel captures shared structural layers, while the embedding channel captures proximity in the relational embedding space. The parameter $\alpha$ balances the two channels.
